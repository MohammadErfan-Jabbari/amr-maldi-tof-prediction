\posterbox[\dimexpr0.30\colheight\relax]{Key Message}{
\RaggedRight
In this Kaggle AMR task, \textbf{domain constraints} (species shift, missing labels, intrinsic resistance) mattered more than exotic models.\\[0.4em]
Our best approach combined \textbf{species-aware validation} with \textbf{rank-averaged ensembles}.\\[0.8em]

}

\posterbox[\dimexpr0.26\colheight\relax]{Dataset \& Challenges}{
\begin{itemize}
  \item Input: \textbf{6,000} binned MALDI-TOF features per isolate (highly sparse).
  \item Output: resistance (0/1) for \textbf{8 antibiotics} (multi-label).
  \item \textbf{Species shift}: train distribution differs sharply from test.
  \item \textbf{Missing labels}: systematic missingness (esp. Amox/Clav) $\Rightarrow$ always mask NaNs.
  \item Metric: \textbf{mean AUC across the 8 antibiotics}.
\end{itemize}
}

\begin{itemize}
  \item \textbf{Shift} $\Rightarrow$ validation split must mimic test species mix.
  \item \textbf{Missingness} $\Rightarrow$ masked objectives, careful pseudo-labeling.
  \item \textbf{Sparsity} $\Rightarrow$ tree models are strong baselines.
\end{itemize}

\posterbox[\dimexpr0.36\colheight\relax]{Methods \& Lessons (What Worked vs Failed)}{
\small
\begin{tabular}{@{}p{0.34\coltextwidth}p{0.58\coltextwidth}@{}}
\toprule
\textbf{Family} & \textbf{Takeaway} \\
\midrule
Baseline (LightGBM) & Strong on sparse MALDI features with NaN masking. \\
PLS + LGB & Helpful dimensionality reduction (better than PCA variants in our runs). \\
Species reweighting & Helps reflect test distribution; not sufficient alone. \\
Ensembling & \textbf{Rank averaging} robust; stacking overfit (great OOF, worse LB). \\
Semi-supervised & Self-training improved validation but did not beat best rank-avg ensemble. \\
\bottomrule
\end{tabular}
\normalsize
\vspace{0.7em}
\begin{itemize}
  \item \textbf{Avoid the trap}: optimize \textbf{mean AUC} across 8 tasks (not a single species metric).
\end{itemize}
}
